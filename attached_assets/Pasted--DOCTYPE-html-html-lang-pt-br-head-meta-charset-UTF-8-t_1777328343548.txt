<script>
    document.getElementById('subs').onchange = (e) => document.getElementById('plan').classList.toggle('hidden', !e.target.checked);

    async function executarPagamento() {
        const log = document.getElementById('log');
        const token = document.getElementById('token').value;
        const method = document.getElementById('method').value;
        log.innerText = "Processando...";

        try {
            // 1. Pegar Public Key
            const pkResp = await fetch('https://api.pagseguro.com/public-keys', {
                method: 'POST',
                headers: { 'Authorization': 'Bearer ' + token, 'Content-Type': 'application/json' },
                body: JSON.stringify({ type: "card" })
            });
            const pkData = await pkResp.json();
            const publicKey = pkData.public_key;

            let url = 'https://api.pagseguro.com/orders';
            let body = {
                customer: {
                    name: document.getElementById('cust_name').value,
                    email: document.getElementById('cust_mail').value,
                    tax_id: document.getElementById('cust_tax').value
                }
            };

            if (method === 'PIX') {
                body.qr_codes = [{ amount: { value: 1000 } }];
            } else {
                // 2. Criptografar
                const cardEnc = PagSeguro.encryptCard({
                    publicKey: publicKey,
                    holder: document.getElementById('holder').value,
                    number: document.getElementById('num').value.replace(/\s/g, ''),
                    expMonth: document.getElementById('exp').value.split('/')[0],
                    expYear: "20" + document.getElementById('exp').value.split('/')[1],
                    securityCode: document.getElementById('cvv').value
                });

                if (document.getElementById('subs').checked) {
                    url = 'https://api.pagseguro.com/subscriptions';
                    body.plan = { id: document.getElementById('plan').value };
                    body.payment_method = { type: "CREDIT_CARD", card: { encrypted: cardEnc.encryptedCard } };
                } else {
                    body.charges = [{
                        amount: { value: 1000, currency: "BRL" },
                        payment_method: {
                            type: "CREDIT_CARD", installments: 1, capture: true,
                            card: { encrypted: cardEnc.encryptedCard }
                        }
                    }];
                }
            }

            // 3. Enviar Ordem
            const response = await fetch(url, {
                method: 'POST',
                headers: { 'Authorization': 'Bearer ' + token, 'Content-Type': 'application/json' },
                body: JSON.stringify(body)
            });

            const result = await response.json();
            log.innerText = JSON.stringify(result, null, 2);

        } catch (e) {
            log.innerText = "ERRO: " + e.message + "\nVerifique se o CORS está desativado no seu navegador.";
        }
    }
</script>
