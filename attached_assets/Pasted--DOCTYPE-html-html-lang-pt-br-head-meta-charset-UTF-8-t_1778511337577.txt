<script>
    async function gerarPix() {
        const btn = document.getElementById('btnPix');
        const status = document.getElementById('status');
        
        btn.disabled = true;
        status.innerText = "Solicitando ao servidor local...";

        const payload = {
            "reference_id": "PEDIDO-" + Date.now(),
            "customer": {
                "name": "Jose Testador",
                "email": "test@sandbox.pagseguro.com.br",
                "tax_id": "12345678909",
                "phone": { "country": "55", "area": "11", "number": "999999999" }
            },
            "items": [{
                "name": "Produto Teste",
                "quantity": 1,
                "unit_amount": 1000
            }],
            "payment_methods": [{ "type": "PIX" }],
            "customer_modifiable": true
        };

        try {
            // Chamamos o NOSSO servidor, não o do PagBank diretamente
            const response = await fetch('http://localhost:3000/criar-checkout', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload)
            });

            const data = await response.json();

            if (data.links) {
                const link = data.links.find(l => l.rel === 'PAY').href;
                status.innerText = "Sucesso! Redirecionando...";
                window.location.href = link;
            } else {
                throw new Error(JSON.stringify(data));
            }
        } catch (err) {
            status.innerHTML = `<span style="color:red">Erro: ${err.message}</span>`;
            btn.disabled = false;
        }
    }
</script>
